def getFinalString(s):
    st = []
    for i in range(len(s)):
        st.append(s[i])
        if len(st)>=3 :
            if st[-3] == 'A' and st[-2] == 'W' and st[-1] == 'S':
                st.pop()
                st.pop()
                st.pop()

    if not len(st) :
        return "-1"
    return "".join(st)


if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    s = input()

    result = getFinalString(s)

    fptr.write(result + '\n')

    fptr.close()

    s = str(num)

#!/bin/python3

import math
import os
import random
import re
import sys


#
# Complete the 'maxLength' function below.
#
# The function is expected to return an INTEGER.
# The function accepts following parameters:
#  1. INTEGER_ARRAY a
#  2. INTEGER k

# 요구 TC : O(N)

# 투 포인터로 찾는다.
def maxLength(arr,k):
    max_len = 0
    sum = 0 # 현재까지 누적 값
    start = 0
    for end in range(len(arr)):
        sum+=arr[end]

        while start <= end and sum > k :
            sum-=arr[start]
            start+=1

        else : #sum <= k : 오른쪽 진행
            max_len = max(max_len,end-start+1)
    return max_len
if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    a_count = int(input().strip())

    a = []

    for _ in range(a_count):
        a_item = int(input().strip())
        a.append(a_item)

    k = int(input().strip())

    result = maxLength(a, k)

    fptr.write(str(result) + '\n')

    fptr.close()

#!/bin/python3

import math
import os
import random
import re
import sys


#
# Complete the 'longestChain' function below.
#
# The function is expected to return an INTEGER.
# The function accepts STRING_ARRAY words as parameter.
#
# 상태트리 : 버리는 것과 버리지않는것. 백트래킹.. 재귀는 시간초과

# max_depth = 0
# 재귀를 최적화 해야한다. -> 정렬을 하지않으면 최적화 하기가 힘들다
# 굳이 재귀를 돌필요없이, 이전에 만든 chain을 그대로 이용한다 dp
# dp사용하면 재귀 필요없음
# 지문에 힌트가있음.. 짧은 문자열부터 돌면서, 딕셔너리에 체인갯수를 저장
#

def longestChain(words):
    dic = dict()
    words = sorted(words,key = lambda x : len(x))
    for word in words:
        dic[word] = 1 # default Value를 설정
        for i in range(len(word)):
            new_word = word[:i] + word[i + 1:]
            if new_word in dic :
                dic[word] = max(dic[word],dic[new_word]+1)
    max_val = 0
    for key,val in dic.items():
        max_val = max(max_val,val)
    return max_val
if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    words_count = int(input().strip())

    words = []

    for _ in range(words_count):
        words_item = input()
        words.append(words_item)

    result = longestChain(words)

    fptr.write(str(result) + '\n')

    fptr.close()
